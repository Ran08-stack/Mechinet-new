// app/(dashboard)/candidates/[id]/page.tsx — redesigned
import { createClient } from "@/lib/supabase/server";
import { notFound } from "next/navigation";
import { formatDate } from "@/lib/utils";
import { FormField, Attachment } from "@/types/database";
import { CandidateHero }   from "@/components/candidates/CandidateHero";
import { StageSelector }   from "@/components/candidates/StageSelector";
import { PersonalInfoCard } from "@/components/candidates/PersonalInfoCard";
import { AnswersCard }      from "@/components/candidates/AnswersCard";
import { AttachmentsCard }  from "@/components/candidates/AttachmentsCard";
import { AISummaryCard }    from "@/components/candidates/AISummaryCard";
import { NotesEditor }      from "@/components/candidates/NotesEditor";
import { ActivityTimeline } from "@/components/candidates/ActivityTimeline";

export default async function CandidatePage({ params }: { params: { id: string } }) {
  const supabase = await createClient();
  const { data: candidate } = await supabase
    .from("candidates").select("*").eq("id", params.id).single();
  if (!candidate) notFound();

  const answers = (candidate.answers ?? {}) as Record<string, string | string[]>;
  const attachments = (candidate.attachments ?? []) as Attachment[];
  let formFields: FormField[] = [];
  if (candidate.form_id) {
    const { data: f } = await supabase.from("forms").select("fields,name").eq("id", candidate.form_id).single();
    if (f && Array.isArray(f.fields)) formFields = f.fields as FormField[];
  }

  return (
    <>
      <CandidateHero candidate={candidate} />
      <StageSelector id={candidate.id} stage={candidate.stage} />

      <div className="grid grid-cols-[minmax(0,1fr)_360px] gap-7 px-8 pb-14 pt-7">
        <div className="space-y-3.5">
          <PersonalInfoCard candidate={candidate} />
          <AnswersCard fields={formFields} answers={answers} />
          <AttachmentsCard attachments={attachments} />
          <NotesEditor candidateId={candidate.id} initialNotes={candidate.notes ?? ""} />
        </div>
        <aside className="sticky top-[70px] flex flex-col gap-3.5">
          <AISummaryCard candidate={candidate} />
          <ActivityTimeline candidateId={candidate.id} />
        </aside>
      </div>
    </>
  );
}
// components/candidates/StageSelector.tsx — redesigned
"use client";
import { useState, useTransition } from "react";
import { useRouter } from "next/navigation";
import { createClient } from "@/lib/supabase/client";
import { type CandidateStage } from "@/types/database";
import { cn } from "@/lib/utils";

const STAGES: { id: CandidateStage; label: string }[] = [
  { id: "new",       label: "חדש" },
  { id: "review",    label: "בבדיקה" },
  { id: "interview", label: "ראיון" },
  { id: "accepted",  label: "התקבל" },
  { id: "rejected",  label: "נדחה" },
];

export function StageSelector({ id, stage }: { id: string; stage: CandidateStage }) {
  const [active, setActive] = useState(stage);
  const [pending, start] = useTransition();
  const router = useRouter();

  async function change(next: CandidateStage) {
    if (next === active || pending) return;
    setActive(next); // optimistic
    start(async () => {
      const supabase = createClient();
      await supabase.from("candidates").update({ stage: next }).eq("id", id);
      router.refresh();
    });
  }

  return (
    <div className="flex items-center gap-3.5 px-8 pb-6">
      <span className="font-mono text-[11px] uppercase tracking-[0.06em] text-fg-subtle">שלב</span>
      <div role="tablist" className="flex flex-1 max-w-[720px] gap-0.5 rounded-full border border-line bg-bg-subtle p-0.5">
        {STAGES.map(s => {
          const on = active === s.id;
          return (
            <button
              key={s.id}
              role="tab"
              aria-selected={on}
              disabled={pending}
              onClick={() => change(s.id)}
              className={cn(
                "flex h-7 flex-1 items-center justify-center gap-1.5 rounded-full text-[13px] font-medium transition-colors disabled:opacity-60",
                on
                  ? `bg-surface shadow-xs text-stage-${s.id}-fg`
                  : "text-fg-muted hover:text-fg"
              )}
            >
              <span
                className="h-1.5 w-1.5 rounded-full"
                style={{ background: `var(--stage-${s.id}-dot)` }}
              />
              {s.label}
            </button>
          );
        })}
      </div>
      <span className="font-mono text-[11.5px] text-fg-subtle whitespace-nowrap">בשלב זה 3 ימים</span>
    </div>
  );
}

"use client";
import { useEffect, useRef, useState } from "react";
import { Check, Bold, Italic, List, Quote } from "lucide-react";
import { createClient } from "@/lib/supabase/client";

export function NotesEditor({ candidateId, initialNotes }: { candidateId: string; initialNotes: string }) {
  const [notes, setNotes] = useState(initialNotes);
  const [savedAt, setSavedAt] = useState<Date | null>(null);
  const timer = useRef<ReturnType<typeof setTimeout>>();

  // Autosave 800ms after last keystroke
  useEffect(() => {
    if (notes === initialNotes) return;
    clearTimeout(timer.current);
    timer.current = setTimeout(async () => {
      const supabase = createClient();
      await supabase.from("candidates").update({ notes }).eq("id", candidateId);
      setSavedAt(new Date());
    }, 800);
    return () => clearTimeout(timer.current);
  }, [notes, candidateId, initialNotes]);

  return (
    <Card title="הערות פנימיות" aside="גלוי לצוות בלבד">
      <Toolbar />
      <textarea
        value={notes}
        onChange={e => setNotes(e.target.value)}
        placeholder="הוסף הערה על המועמד… השתמשו ב־@ לאזכור חבר צוות."
        className="min-h-[130px] w-full resize-y rounded-md border border-line bg-surface px-4 py-3.5 text-[13px] leading-7 outline-none focus:border-accent focus:shadow-focus"
        dir="rtl"
      />
      <footer className="mt-2.5 flex items-center justify-between text-[13px] text-fg-subtle">
        {savedAt && (
          <span className="flex items-center gap-1 text-[var(--success)]">
            <Check className="h-3 w-3" /> נשמר אוטומטית · {timeAgo(savedAt)}
          </span>
        )}
        <span className="font-mono text-[11px]">{notes.length} תווים</span>
      </footer>
    </Card>
  );
}
