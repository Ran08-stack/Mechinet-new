// app/(dashboard)/candidates/page.tsx
import { Search, SlidersHorizontal, ArrowDownUp, LayoutGrid, Download, Plus,
         MoreHorizontal, ChevronLeft, ChevronRight } from "lucide-react";
import { StageBadge, type Stage } from "@/components/StageBadge";
import { getCandidates, getStageCounts } from "@/lib/candidates";

const stages: { id: Stage | "all"; label: string }[] = [
  { id: "all",       label: "הכל" },
  { id: "new",       label: "חדש" },
  { id: "review",    label: "בבדיקה" },
  { id: "interview", label: "ראיון" },
  { id: "accepted",  label: "התקבל" },
  { id: "rejected",  label: "נדחה" },
];

export default async function CandidatesPage({ searchParams }: {
  searchParams: { stage?: Stage; q?: string; page?: string };
}) {
  const [rows, counts] = await Promise.all([
    getCandidates({ stage: searchParams.stage, q: searchParams.q, page: +(searchParams.page ?? 1) }),
    getStageCounts(),
  ]);
  const currentStage = searchParams.stage ?? "all";

  return (
    <>
      {/* Heading */}
      <header className="flex items-end justify-between gap-6 px-7 pb-4.5 pt-7">
        <div>
          <h1 className="text-2xl font-semibold tracking-tight">מועמדים</h1>
          <p className="mt-1.5 text-sm text-fg-muted">{counts.all} מועמדים פעילים · עודכן לפני 4 דקות</p>
        </div>
        <StatsStrip counts={counts} />
      </header>

      {/* Stage tabs */}
      <nav className="flex items-center gap-1 overflow-x-auto border-b border-line bg-surface px-7">
        {stages.map(s => {
          const active = currentStage === s.id;
          return (
            <Link
              key={s.id}
              href={`/candidates?stage=${s.id}`}
              className={cn(
                "-mb-px inline-flex items-center gap-2 border-b-2 px-3 py-3 text-[13px] font-medium transition-colors",
                active
                  ? "border-accent text-fg"
                  : "border-transparent text-fg-muted hover:text-fg"
              )}
            >
              {s.id !== "all" && (
                <span
                  className="h-1.5 w-1.5 rounded-full"
                  style={{ background: `var(--stage-${s.id}-dot)` }}
                />
              )}
              {s.label}
              <span className={cn(
                "rounded-full border px-1.5 py-px font-mono text-[10.5px]",
                active
                  ? "border-accent-line bg-accent-soft text-accent-hover"
                  : "border-line bg-bg-muted text-fg-subtle"
              )}>{counts[s.id]}</span>
            </Link>
          );
        })}
      </nav>

      {/* Toolbar */}
      <div className="flex flex-wrap items-center gap-2.5 bg-bg px-7 py-3.5">
        <SearchInput defaultValue={searchParams.q} placeholder="חיפוש לפי שם, אימייל, או עיר…" />
        <PillButton icon={SlidersHorizontal}>סינון נוסף</PillButton>
        <PillButton icon={ArrowDownUp}>מיון: תאריך הגשה</PillButton>
        <div className="flex-1" />
        <span className="me-2 font-mono text-[11.5px] text-fg-subtle">{rows.length} מתוך {counts.all}</span>
        <PillButton icon={LayoutGrid}>תצוגה</PillButton>
        <PillButton icon={Download}>ייצוא</PillButton>
      </div>

      {/* Table */}
      <div className="mx-7 overflow-hidden rounded-lg border border-line bg-surface">
        <table className="w-full text-[13px]">
          <thead className="bg-surface-2">
            <tr className="[&_th]:border-b [&_th]:border-line [&_th]:px-3.5 [&_th]:py-2.5 [&_th]:text-start [&_th]:font-mono [&_th]:text-[11px] [&_th]:uppercase [&_th]:tracking-[0.06em] [&_th]:text-fg-subtle">
              <th className="w-9"><Checkbox indeterminate={"all"} /></th>
              <th>שם</th><th>אימייל</th><th>עיר</th>
              <th>שלב</th><th>מקור</th><th>תאריך הגשה</th><th className="w-9" />
            </tr>
          </thead>
          <tbody>
            {rows.map(r => (
              <tr
                key={r.id}
                onClick={() => router.push(`/candidates/${r.id}`)}
                className="group cursor-pointer border-b border-line-faint last:border-0 transition-colors hover:bg-bg-subtle data-[selected=true]:bg-accent-soft"
                data-selected={r.selected}
              >
                <td className="px-3.5 py-2.5" onClick={e => e.stopPropagation()}><Checkbox checked={r.selected} /></td>
                <td className="px-3.5 py-2.5"><NameCell candidate={r} /></td>
                <td className="px-3.5 py-2.5 font-mono text-[12px] text-fg-muted [direction:ltr] text-start">{r.email}</td>
                <td className="px-3.5 py-2.5 text-fg-muted">{r.city}</td>
                <td className="px-3.5 py-2.5"><StageBadge stage={r.stage} /></td>
                <td className="px-3.5 py-2.5"><SourceChip source={r.source} /></td>
                <td className="px-3.5 py-2.5 tabular-nums text-fg-muted text-[12.5px]">{formatDate(r.appliedAt)}</td>
                <td className="w-9 px-2 text-center">
                  <button className="grid h-6.5 w-6.5 place-items-center rounded-sm text-fg-faint opacity-0 group-hover:opacity-100 hover:bg-bg-muted hover:text-fg">
                    <MoreHorizontal className="h-[15px] w-[15px]" />
                  </button>
                </td>
              </tr>
            ))}
          </tbody>
        </table>
      </div>

      <Pagination total={counts.all} pageSize={12} page={1} />
    </>
  );
}

export type Stage = "new" | "review" | "interview" | "accepted" | "rejected";

const labels: Record<Stage, string> = {
  new: "חדש",
  review: "בבדיקה",
  interview: "ראיון",
  accepted: "התקבל",
  rejected: "נדחה",
};

export function StageBadge({ stage }: { stage: Stage }) {
  return (
    <span
      className="inline-flex h-[22px] items-center gap-1.5 whitespace-nowrap rounded-full border px-2.5 text-[11.5px] font-medium leading-none"
      style={{
        background:  `var(--stage-${stage}-bg)`,
        color:       `var(--stage-${stage}-fg)`,
        borderColor: `var(--stage-${stage}-line)`,
      }}
    >
      <span
        className="h-1.5 w-1.5 rounded-full"
        style={{ background: `var(--stage-${stage}-dot)` }}
      />
      {labels[stage]}
    </span>
  );
}
